303 } -> std::same_as<bool>;

305 })
306 static constexpr bool impl(char_t<alph2_t> const chr, meta::priority_tag<1>) noexcept
307 {

412struct assign_char_strictly_to_fn
413{
415 template <typename alph_t>
417 requires(requires(alph_t a, bio::alphabet::char_t<alph_t> r) {
418 {bio::alphabet::assign_char_to(r, a)};
419 {bio::alphabet::char_is_valid_for<alph_t>(r)};
420 })
422 auto operator()(bio::alphabet::char_t<alph_t> const r, alph_t && a) const -> alph_t
423 {
424 if (!bio::alphabet::char_is_valid_for<alph_t>(r))
425 throw bio::alphabet::invalid_char_assignment{meta::detail::type_name_as_string<alph_t>, r};

427 return bio::alphabet::assign_char_to(r, a);
