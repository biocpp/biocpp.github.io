500
503struct assign_char_strictly_to_fn
504{
506 template <typename alph_t>
508 requires(requires(alph_t a, bio::alphabet::char_t<alph_t> r) {
509 {
510 bio::alphabet::assign_char_to(r, a)
511 } -> std::convertible_to<alph_t>;
512 {
513 bio::alphabet::char_is_valid_for<alph_t>(r)
514 } -> std::same_as<bool>;
515 })
517 auto operator()(bio::alphabet::char_t<alph_t> const r, alph_t && a) const -> alph_t
518 {
519 if (!bio::alphabet::char_is_valid_for<alph_t>(r))
520 throw bio::alphabet::invalid_char_assignment{meta::detail::type_name_as_string<alph_t>, r};
521
522 return bio::alphabet::assign_char_to(r, a);
523 }
524};
