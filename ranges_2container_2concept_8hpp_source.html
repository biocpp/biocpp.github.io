48template <typename type>
49concept container = requires(type val, type val2, type const cval, typename type::iterator it)
50{
51 // member types
52 typename type::value_type;
53 typename type::reference;
54 typename type::const_reference;
55 /*
56 typename type::iterator;
57 requires std::forward_iterator<typename type::iterator>;
58 // NOTE check whether iterator is const convertible
59 { it } -> std::same_as<typename type::const_iterator>;
60
61 typename type::const_iterator;
62 requires std::forward_iterator<typename type::const_iterator>;
63
64 typename type::difference_type;
65 typename type::size_type;
66 requires std::is_same_v<
67 typename type::difference_type,
68 typename std::iterator_traits<typename type::iterator>::difference_type
69 >;
70 requires std::is_same_v<
71 typename std::iterator_traits<typename type::iterator>::difference_type,
72 typename std::iterator_traits<typename type::const_iterator>::difference_type
73 >;
74*/
75
76 // clang-format off
77 // methods and operator
78 { type{} } -> std::same_as<type>; // default constructor
79 { type{type{}} } -> std::same_as<type>; // copy/move constructor
80 { val = val2 } -> std::same_as<type &>; // assignment
81 { (&val)->~type() }; // destructor
82
83 { val.begin() } -> std::same_as<typename type::iterator>;
84 { val.end() } -> std::same_as<typename type::iterator>;
85 { cval.begin() } -> std::same_as<typename type::const_iterator>;
86 { cval.end() } -> std::same_as<typename type::const_iterator>;
87 { val.cbegin() } -> std::same_as<typename type::const_iterator>;
88 { val.cend() } -> std::same_as<typename type::const_iterator>;
89
90 requires !std::equality_comparable<typename type::value_type> || std::equality_comparable<type>;
91
92 { val.swap(val2) } -> std::same_as<void>;
93 { swap(val, val2) } -> std::same_as<void>;
94 { std::swap(val, val2) } -> std::same_as<void>;
95
96 { val.size() } -> std::same_as<typename type::size_type>;
97 { val.max_size() } -> std::same_as<typename type::size_type>;
98 { val.empty() } -> std::same_as<bool>;
99 // clang-format on
100};

116template <typename type>
117concept sequence_container = requires(type val, type val2, type const cval)
118{
119 requires container<type>;
120
121 // clang-format off
122 // construction
123 { type{typename type::size_type{}, typename type::value_type{}} };
124 { type{val2.begin(), val2.end()} }; // NOTE that this could be any input iterator:
125 { type{std::initializer_list<typename type::value_type>{}} };
126 { val = std::initializer_list<typename type::value_type>{} } -> std::same_as<type &>;
127
128 // assignment NOTE return type is type & for std::string and void for other stl containers:
129 { val.assign(val2.begin(), val2.end()) };
130 { val.assign(std::initializer_list<typename type::value_type>{}) };
131 { val.assign(typename type::size_type{}, typename type::value_type{}) };
132
133 // modify container
134 // TODO: how do you model this?
135 // BIOCPP_RETURN_TYPE_CONSTRAINT(val.emplace(typename type::const_iterator{}, ?),
136 // std::same_as, typename type::iterator);
137 { val.insert(val.cbegin(), val2.front()) } -> std::same_as<typename type::iterator>;
138 { val.insert(val.cbegin(), typename type::value_type{}) } -> std::same_as<typename type::iterator>;
139 { val.insert(val.cbegin(), typename type::size_type{}, typename type::value_type{}) }
140 -> std::same_as<typename type::iterator>;
141 { val.insert(val.cbegin(), val2.begin(), val2.end()) } -> std::same_as<typename type::iterator>;
142 { val.insert(val.cbegin(), std::initializer_list<typename type::value_type>{}) } ->
143 std::same_as<typename type::iterator>;
144
145 { val.erase(val.cbegin()) } -> std::same_as<typename type::iterator>;
146 { val.erase(val.cbegin(), val.cend()) } -> std::same_as<typename type::iterator>;
147
148 { val.push_back(val.front()) } -> std::same_as<void>;
149 { val.push_back(typename type::value_type{}) } -> std::same_as<void>;
150 { val.pop_back() } -> std::same_as<void>;
151 { val.clear() } -> std::same_as<void>;
152
153 // access container
154 { val.front() } -> std::same_as<typename type::reference>;
155 { cval.front() } -> std::same_as<typename type::const_reference>;
156 { val.back() } -> std::same_as<typename type::reference>;
157 { cval.back() } -> std::same_as<typename type::const_reference>;
158 // clang-format on
159};

177template <typename type>
178concept random_access_container = requires(type val)
179{
180 requires sequence_container<type>;
181
182 // clang-format off
183 // access container
184 { val[0] } -> std::same_as<typename type::reference>;
185 { val.at(0) } -> std::same_as<typename type::reference>;
186
187 // modify container
188 { val.resize(0) } -> std::same_as<void>;
189 { val.resize(0, typename type::value_type{}) } -> std::same_as<void>;
