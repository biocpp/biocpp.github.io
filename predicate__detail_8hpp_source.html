39 template <size_t N>
40 class constexpr_pseudo_bitset : public std::array<bool, N>
41 {
42 private:
44  using base_t = std::array<bool, N>;
45 public:
47  using base_t::base_t;
48 
50  constexpr constexpr_pseudo_bitset operator|(constexpr_pseudo_bitset rhs) const noexcept
51  {
52  for (size_t i = 0; i < N; ++i)
53  rhs[i] = rhs[i] || base_t::operator[](i);
54 
55  return rhs;
56  }
58  constexpr constexpr_pseudo_bitset operator~() const noexcept
59  {
60  constexpr_pseudo_bitset ret{};
61  for (size_t i = 0; i < N; ++i)
62  ret[i] = !base_t::operator[](i);
63 
64  return ret;
65  }
66 };

109 concept char_predicate = requires
110 {

162 template <typename derived_t>
163 struct char_predicate_base
164 {
166  using data_t = constexpr_pseudo_bitset<257>; // sizeof(char) plus EOF
167 
173  template <char_predicate rhs_t>
174  constexpr auto operator||(rhs_t const &) const
175  {
176  return char_predicate_combiner<derived_t, rhs_t>{};
177  }
178 
180  constexpr auto operator!() const
181  {
182  return char_predicate_negator<derived_t>{};
183  }
185 
190  template <std::integral value_t>
191  constexpr bool operator()(value_t const val) const noexcept
193  requires (sizeof(value_t) == 1)
195  {
196  return derived_t::data[static_cast<unsigned char>(val)];
197  }
198 
200  template <std::integral value_t>
201  constexpr bool operator()(value_t const val) const noexcept
203  requires (sizeof(value_t) != 1)
205  {
206  using char_trait = std::char_traits<value_t>;
207  return (static_cast<std::make_unsigned_t<value_t>>(val) < 256) ? operator()(static_cast<uint8_t>(val)) :
208  (char_trait::eq_int_type(val, char_trait::eof())) ? derived_t::data[256] : false;
209  }
211 
216  std::string message() const
217  {
218  return derived_t::msg;
219  }
