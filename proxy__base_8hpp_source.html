188 constexpr operator alphabet_type() const noexcept
189 {
190 return alphabet::assign_rank_to(static_cast<derived_type const &>(*this).to_rank(), alphabet_type{});
191
192 /* Instead of static_cast'ing to the alphabet_type which also considers the constructors of the alphabet_type,
193 * we explicitly invoke this operator in various places.
194 * This prevents errors associated with using alphabet_type's constructors.
195 *
196 * This is one of error cases:
197 * The tuple composite bio::alphabet::qualified returns a component_proxy which inherits from proxy_base_base.
198 * The qualified alphabet itself inherits from quality_base.
199 * Now when accessing get<1>(seq_qual_alph) we want to call to_phred at some point because we want the quality,
200 * therefore the to_phred function from proxy_base is called, but this function did a static_cast to the
201 * derived type which is calling the constructor from quality_base. Unfortunately now, the generic quality_base
202 * constructor uses `assign_phred_to(to_phred(other), static_cast<derived_type &>(*this))`; (here) which again
203 * tries to call to_phred of the proxy_base => infinite loop :boom:
204 */
205 }
