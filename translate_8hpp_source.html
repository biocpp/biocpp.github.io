59template <bool single>
60struct translate_fn
61{
63 static constexpr alphabet::translation_frames default_frames =
64 single ? alphabet::translation_frames::FWD_FRAME_0 : alphabet::translation_frames::SIX_FRAME;
65
67 constexpr auto operator()(alphabet::translation_frames const tf = default_frames) const
68 {
69 return detail::adaptor_from_functor{*this, tf};
70 }

77 template <std::ranges::range urng_t>
78 constexpr auto operator()(urng_t && urange, alphabet::translation_frames const tf = default_frames) const
79 {
80 static_assert(std::ranges::viewable_range<urng_t>,
81 "The range parameter to views::translate[_single] cannot be a temporary of a non-view range.");
82 static_assert(std::ranges::sized_range<urng_t>,
83 "The range parameter to views::translate[_single] must model std::ranges::sized_range.");
84 static_assert(std::ranges::random_access_range<urng_t>,
85 "The range parameter to views::translate[_single] must model std::ranges::random_access_range.");
86 static_assert(alphabet::nucleotide_alphabet<std::ranges::range_reference_t<urng_t>>,
87 "The range parameter to views::translate[_single] must be over elements of "
88 "bio::alphabet::alphabet::nucleotide_alphabet.");
89
90 if constexpr (single)
91 return detail::view_translate_single{std::forward<urng_t>(urange), tf};
92 else
93 return detail::view_translate{std::forward<urng_t>(urange), tf};
94 }

97 template <std::ranges::range urng_t>
98 constexpr friend auto operator|(urng_t && urange, translate_fn const & me)
99 {
100 return me(std::forward<urng_t>(urange));
101 }

251 size_type size()
252 {
253 switch (tf)
254 {
255 case alphabet::translation_frames::FWD_FRAME_0:
256 [[fallthrough]];
257 case alphabet::translation_frames::REV_FRAME_0:
258 return std::ranges::size(urange) / 3;

261 [[fallthrough]];

264 break;
265 case alphabet::translation_frames::FWD_FRAME_2:
266 [[fallthrough]];
267 case alphabet::translation_frames::REV_FRAME_2:
268 return (std::max<size_type>(std::ranges::size(urange), 2) - 2) / 3;

541 view_translate(urng_t _urange, alphabet::translation_frames const _tf = alphabet::translation_frames::SIX_FRAME) :
542 urange{std::move(_urange)}, tf{_tf}
543 {
544 if ((_tf & alphabet::translation_frames::FWD_FRAME_0) == alphabet::translation_frames::FWD_FRAME_0)
545 selected_frames.push_back(alphabet::translation_frames::FWD_FRAME_0);
546 if ((_tf & alphabet::translation_frames::FWD_FRAME_1) == alphabet::translation_frames::FWD_FRAME_1)
547 selected_frames.push_back(alphabet::translation_frames::FWD_FRAME_1);
548 if ((_tf & alphabet::translation_frames::FWD_FRAME_2) == alphabet::translation_frames::FWD_FRAME_2)
549 selected_frames.push_back(alphabet::translation_frames::FWD_FRAME_2);
550 if ((_tf & alphabet::translation_frames::REV_FRAME_0) == alphabet::translation_frames::REV_FRAME_0)
551 selected_frames.push_back(alphabet::translation_frames::REV_FRAME_0);
552 if ((_tf & alphabet::translation_frames::REV_FRAME_1) == alphabet::translation_frames::REV_FRAME_1)
553 selected_frames.push_back(alphabet::translation_frames::REV_FRAME_1);
554 if ((_tf & alphabet::translation_frames::REV_FRAME_2) == alphabet::translation_frames::REV_FRAME_2)
555 selected_frames.push_back(alphabet::translation_frames::REV_FRAME_2);
556 }
