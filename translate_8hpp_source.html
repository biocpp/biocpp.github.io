101  static constexpr translation_frames default_frames = single ?

103  translation_frames::SIX_FRAME;

106  constexpr auto operator()(translation_frames const tf = default_frames) const
107  {
108  return detail::adaptor_from_functor{*this, tf};
109  }
110 
116  template <std::ranges::range urng_t>
117  constexpr auto operator()(urng_t && urange, translation_frames const tf = default_frames) const
118  {
119  static_assert(std::ranges::viewable_range<urng_t>,
120  "The range parameter to views::translate[_single] cannot be a temporary of a non-view range.");
121  static_assert(std::ranges::sized_range<urng_t>,
122  "The range parameter to views::translate[_single] must model std::ranges::sized_range.");
123  static_assert(std::ranges::random_access_range<urng_t>,
124  "The range parameter to views::translate[_single] must model std::ranges::random_access_range.");
125  static_assert(nucleotide_alphabet<std::ranges::range_reference_t<urng_t>>,
126  "The range parameter to views::translate[_single] must be over elements of bio::nucleotide_alphabet.");
127 
128  if constexpr (single)
129  return detail::view_translate_single{std::forward<urng_t>(urange), tf};
130  else
131  return detail::view_translate{std::forward<urng_t>(urange), tf};
132  }
