53 template <genetic_code gc = genetic_code::CANONICAL, nucleotide_alphabet nucl_type>
54 constexpr aa27 translate_triplet(nucl_type const & n1, nucl_type const & n2, nucl_type const & n3) noexcept
55 {
56  if constexpr (std::same_as<nucl_type, dna4> || std::same_as<nucl_type, dna5> || std::same_as<nucl_type, dna15>)
57  {
58  // table exists for dna15 and is generated for dna4 and dna5 (compile time ok, because small)
59  return bio::detail::translation_table<nucl_type, gc>::VALUE[to_rank(n1)][to_rank(n2)][to_rank(n3)];
60  }
61  else if constexpr (std::same_as<nucl_type, rna4> || std::same_as<nucl_type, rna5> || std::same_as<nucl_type, rna15>)
62  {
63  using rna2dna_t = std::conditional_t<std::same_as<nucl_type, rna4>, dna4,
64  std::conditional_t<std::same_as<nucl_type, rna5>, dna5,
65  std::conditional_t<std::same_as<nucl_type, rna15>, dna15, void>>>;
66 
67  // we can use dna's tables, because ranks are identical
68  return bio::detail::translation_table<rna2dna_t, gc>::VALUE[to_rank(n1)][to_rank(n2)][to_rank(n3)];
69  }
70  else // composites or user defined nucleotide
71  {
72  // we cast to dna15; slightly slower run-time, but lot's of compile time saved for large alphabets.
73  // (nucleotide types can be converted to dna15 by definition)
74  return bio::detail::translation_table<dna15, gc>::VALUE[to_rank(static_cast<dna15>(n1))]
75  [to_rank(static_cast<dna15>(n2))]
76  [to_rank(static_cast<dna15>(n3))];
77  }
78 }
