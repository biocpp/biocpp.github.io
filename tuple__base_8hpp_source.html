448 template <size_t index>
449 constexpr rank_type to_component_rank() const noexcept
450 {
451 if constexpr (alphabet_size < 1024) // computation is cached for small alphabets
452 {
453 return rank_to_component_rank[index][to_rank()];
454 }
455 else
456 {
457 return (to_rank() / cummulative_alph_sizes[index]) %
458 bio::alphabet::size<meta::detail::pack_traits::at<index, component_types...>>;
459 }
460 }
461
463 template <size_t index>
464 constexpr void assign_component_rank(ptrdiff_t const r) noexcept
465 {
466 assign_rank(static_cast<ptrdiff_t>(to_rank()) + ((r - static_cast<ptrdiff_t>(to_component_rank<index>())) *
467 static_cast<ptrdiff_t>(cummulative_alph_sizes[index])));
468 }
469
471 static constexpr std::array<rank_type, component_list::size()> cummulative_alph_sizes = []() constexpr
472 {
473 // create array (1, |sigma1|, |sigma1|*|sigma2|, ... , |sigma1|*...*|sigmaN|)
474 std::array<rank_type, component_list::size() + 1> ret{};
475 ret[0] = 1;
476 size_t count = 1;
477 using reverse_list_t = decltype(meta::list_traits::detail::reverse(component_list{}));
478 bio::meta::detail::for_each<reverse_list_t>([&](auto alphabet_type_identity) constexpr {
479 using alphabet_t = typename decltype(alphabet_type_identity)::type;
480 ret[count] = static_cast<rank_type>(bio::alphabet::size<alphabet_t> * ret[count - 1]);
481 ++count;
482 });
483
484 // reverse and strip one: (|sigma1|*...*|sigmaN-1|, ... |sigma1|*|sigma2|, |sigma1|, 1)
485 // reverse order guarantees that the first alphabet is the most significant rank contributer
486 // resulting in element-wise alphabetical ordering on comparison
487 std::array<rank_type, component_list::size()> ret2{};
488 for (size_t i = 0; i < component_list::size(); ++i)
489 ret2[i] = ret[component_list::size() - i - 1];
490
491 return ret2;
492 }
493 ();

496 template <std::size_t... idx>
497 static constexpr rank_type rank_sum_helper(component_types... components,
498 std::index_sequence<idx...> const &) noexcept
499 {
500 return ((bio::alphabet::to_rank(components) * cummulative_alph_sizes[idx]) + ...);
501 }
502
504 static constexpr std::array < std::array<rank_type,
505 alphabet_size<1024 ? alphabet_size : 0>, // not for big alphs
506 meta::list_traits::size<component_list>>
507 rank_to_component_rank = []() constexpr
508 {
509 std::array < std::array<rank_type,
510 alphabet_size<1024 ? alphabet_size : 0>, // not for big alphs
511 meta::list_traits::size<component_list>>
512 ret{};
513
514 if constexpr (alphabet_size < 1024)
515 {
516 std::array<size_t, alphabet_size> alph_sizes{bio::alphabet::size<component_types>...};
517
518 for (size_t i = 0; i < meta::list_traits::size<component_list>; ++i)
519 for (size_t j = 0; j < static_cast<size_t>(alphabet_size); ++j)
520 ret[i][j] = (j / cummulative_alph_sizes[i]) % alph_sizes[i];
521 }
522
523 return ret;
524 }
525 ();

539class tuple_base<derived_type, component_types...>::component_proxy :
540 public proxy_base<component_proxy<alphabet_type, index>, alphabet_type>
