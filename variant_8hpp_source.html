375 template <std::same_as<variant> variant_t, typename indirect_alternative_type>
377 requires((detail::variant_general_guard<indirect_alternative_type, alternative_types...>)&&(
378 meta::weakly_equality_comparable_with<indirect_alternative_type, alternative_types> || ...))
380 friend constexpr bool operator==(variant_t const lhs, indirect_alternative_type const rhs) noexcept
381 {
382 using alternative_predicate = detail::weakly_equality_comparable_with_<indirect_alternative_type>;
383 constexpr auto alternative_position =
384 meta::list_traits::find_if<alternative_predicate::template invoke, alternatives>;
385 using alternative_t = meta::list_traits::at<alternative_position, alternatives>;
386 return lhs.template holds_alternative<alternative_t>() &&
387 (lhs.template convert_unsafely_to<alternative_t>() == rhs);
388 }

407 template <size_t index, bool throws>
408 constexpr auto convert_impl() const noexcept(!throws) -> meta::list_traits::at<index, alternatives>
409 {
410 static_assert(index < alphabet_size, "The variant contains less alternatives than you are checking.");
411 using alternative_t = meta::list_traits::at<index, alternatives>;
412
413 if constexpr (throws)
414 {
415 if (!holds_alternative<index>()) // [[unlikely]]
416 {
417 throw std::bad_variant_access{};
418 }
419 }
420
421 return bio::alphabet::assign_rank_to(to_rank() - partial_sum_sizes[index], alternative_t{});
422 }

431 static constexpr std::array<rank_type, sizeof...(alternative_types) + 1> partial_sum_sizes = []() constexpr
432 {
433 constexpr size_t N = sizeof...(alternative_types) + 1;
434
435 std::array<rank_type, N> partial_sum{0, bio::alphabet::size<alternative_types>...};
436 for (size_t i = 1u; i < N; ++i)
437 partial_sum[i] += partial_sum[i - 1];
438
439 return partial_sum;
440 }

449 static constexpr std::array<char_type, alphabet_size> rank_to_char = []() constexpr
450 {
451 // Explicitly writing assign_rank_to_char within assign_rank_to_char
452 // causes this bug (g++-7 and g++-8):
453 // https://gcc.gnu.org/bugzilla/show_bug.cgi?id=84684
454 auto assign_rank_to_char = [](auto alternative, size_t rank) constexpr
455 {
456 return bio::alphabet::to_char(bio::alphabet::assign_rank_to(rank, alternative));
457 };

459 auto assign_value_to_char = [assign_rank_to_char](auto alternative,
460 auto & value_to_char,
461 auto & value) constexpr
462 {
463 using alternative_t = std::decay_t<decltype(alternative)>;
464 for (size_t i = 0u; i < bio::alphabet::size<alternative_t>; ++i, ++value)
465 value_to_char[value] = assign_rank_to_char(alternative, i);
466 };
467
468 unsigned value = 0u;
469 std::array<char_type, alphabet_size> value_to_char{};
470
471 // initializer lists guarantee sequencing;
472 // the following expression behaves as:
473 // for(auto alternative: alternative_types)
474 // assign_rank_to_char(alternative, rank_to_char, value);
475 ((assign_value_to_char(alternative_types{}, value_to_char, value)), ...);
476
477 return value_to_char;
478 }

511 static constexpr auto first_valid_char_table = []() constexpr
512 {
513 constexpr size_t alternative_size = sizeof...(alternative_types);
514 constexpr size_t table_size = meta::detail::size_in_values_v<char_type>;
515 using first_alphabet_t = meta::detail::min_viable_uint_t<alternative_size>;
516
517 std::array<first_alphabet_t, table_size> lookup_table{};
518
519 for (size_t i = 0u; i < table_size; ++i)
520 {
521 char_type chr = static_cast<char_type>(i);
522
523 std::array<bool, alternative_size> valid_chars{char_is_valid_for<alternative_types>(chr)...};
524
525 auto found_it = std::find(valid_chars.begin(), valid_chars.end(), true);
526 lookup_table[i] = found_it - valid_chars.begin();
527 }
528
529 return lookup_table;
530 }

539 static constexpr std::array<rank_type, meta::detail::size_in_values_v<char_type>> char_to_rank = []() constexpr
540 {
541 constexpr size_t alternative_size = sizeof...(alternative_types);
542 constexpr size_t table_size = meta::detail::size_in_values_v<char_type>;
543
544 std::array<rank_type, table_size> char_to_rank{};
545
546 for (size_t i = 0u; i < table_size; ++i)
547 {
548 char_type chr = static_cast<char_type>(i);
549
550 std::array<rank_type, alternative_size> ranks{rank_by_type_(assign_char_to(chr, alternative_types{}))...};
551
552 // if no char_is_valid_for any alternative use the rank of the first alternative
553 char_to_rank[i] = first_valid_char_table[i] < alternative_size ? ranks[first_valid_char_table[i]] : 0;
554 }
555
556 return char_to_rank;
557 }
