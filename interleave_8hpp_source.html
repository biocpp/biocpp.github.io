263 template <std::ranges::range urng_t, std::ranges::range inserted_rng_t, std::integral size_type>
264 constexpr auto operator()(urng_t && urange, size_type const size, inserted_rng_t && i) const noexcept
265 {
266 static_assert(std::ranges::random_access_range<urng_t>,
267 "The underlying range parameter in views::interleave must model "
268 "std::ranges::random_access_range.");
269 static_assert(std::ranges::viewable_range<urng_t>,
270 "The underlying range parameter in views::interleave must model std::ranges::viewable_range.");
271 static_assert(std::ranges::sized_range<urng_t>,
272 "The underlying range parameter in views::interleave must model std::ranges::sized_range.");
273
274 static_assert(std::ranges::random_access_range<inserted_rng_t>,
275 "The range to be inserted by views::interleave must model std::ranges::forward_range.");
276 static_assert(std::ranges::sized_range<inserted_rng_t>,
277 "The range to be inserted by views::interleave must model std::ranges::sized_range.");
278
279 return detail::view_interleave{std::forward<urng_t>(urange),
280 static_cast<size_t>(size),
281 std::forward<inserted_rng_t>(i)};
282 }
283};
