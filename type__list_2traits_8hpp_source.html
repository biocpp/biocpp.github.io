102 template <size_t count, typename t>

148 template <typename ...current_list_t, typename remove_t, typename ...remove_list_t>
149 constexpr auto type_list_difference(bio::type_list<current_list_t...>, bio::type_list<remove_t, remove_list_t...>)
150 {
151  constexpr auto pos = bio::pack_traits::find<remove_t, current_list_t...>;
152  if constexpr (pos >= 0)
153  {
154  using split_list_t = bio::pack_traits::split_after<pos, current_list_t...>;
155 
156  using split_list1_t = typename split_list_t::first_type;
157  using split_list2_t = decltype(drop_front(typename split_list_t::second_type{}));
158  using filtered_list_t = decltype(concat(split_list1_t{}, split_list2_t{}));
159  return type_list_difference(filtered_list_t{}, bio::type_list<remove_t, remove_list_t...>{});
160  }
161  else
162  {
163  // remove_t not contained in current_list_t
164  using filtered_list_t = bio::type_list<current_list_t...>;
165  return type_list_difference(filtered_list_t{}, bio::type_list<remove_list_t...>{});
166  }
