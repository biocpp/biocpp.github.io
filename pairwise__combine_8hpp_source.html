89 explicit constexpr pairwise_combine_view(underlying_range_type range) : u_range{std::move(range)}
90 {
91 // Check if range is empty.
92 if (std::ranges::empty(u_range))
93 {
94 back_iterator = std::ranges::end(u_range);
95 }
96 else
97 {
98 if constexpr (std::ranges::bidirectional_range<underlying_range_type>)
99 { // Simply take one before the end. We can do this as we require underlying_range_type to be a common range.
100 back_iterator = std::ranges::prev(std::ranges::end(u_range));
101 }
102 else
103 { // For all other cases we need to set the back_iterator in linear time to the correct position.
104 back_iterator = std::ranges::begin(u_range);
105 if constexpr (std::ranges::sized_range<underlying_range_type>)
106 {
107 std::ranges::advance(back_iterator, std::ranges::size(u_range) - 1);
108 }
109 else // We don't have the size, so we need to increment until one before the end in a linear pass.
110 {
111 auto tmp_it = back_iterator;
112 do
113 {
114 back_iterator = tmp_it;
115 }
116 while (++tmp_it != std::ranges::end(u_range));
117 }
118 }
119 }
120 }

517 constexpr size_t to_index() const
518 noexcept(noexcept(std::declval<underlying_iterator_type &>() - std::declval<underlying_iterator_type &>()))
520 requires std::random_access_iterator<underlying_iterator_type>
522 {
523 size_t src_size = end_it - begin_it;
524 size_t index_i = first_it - begin_it;
525 size_t index_j = second_it - begin_it;
526 return (src_size * (src_size - 1) / 2) - (src_size - index_i) * ((src_size - index_i) - 1) / 2 + index_j -
527 index_i - 1;
528 }

534 constexpr void from_index(size_t const index) noexcept(
535 noexcept(std::declval<underlying_iterator_type &>() -
536 std::declval<underlying_iterator_type &>()) && noexcept(std::declval<underlying_iterator_type &>() + 1))
538 requires std::random_access_iterator<underlying_iterator_type>
540 {
541 size_t src_size = end_it - begin_it;
542 size_t index_i =
543 src_size - 2 - static_cast<size_t>(std::sqrt(-8 * index + 4 * src_size * (src_size - 1) - 7) / 2.0 - 0.5);
544 size_t index_j =
545 index + index_i + 1 - src_size * (src_size - 1) / 2 + (src_size - index_i) * ((src_size - index_i) - 1) / 2;
546 first_it = begin_it + index_i;
547 second_it = begin_it + index_j;
548 }
