958  template <std::ranges::forward_range rng_type>
959  iterator insert(const_iterator pos, size_type const count, rng_type && value)
961  requires is_compatible_with_value_type<rng_type>
963  {
964  auto const pos_as_num = std::distance(cbegin(), pos); // we want to insert BEFORE this position
965  // TODO SEQAN_UNLIKELY
966  if (count == 0)
967  return begin() + pos_as_num;
968 
969  /* TODO implement views::flat_repeat_n that is like
970  * views::repeat_n(value, count) | views::join | ranges::views::bounded;
971  * but preserves random access and size.
972  *
973  * then do
974  * auto concatenated = ranges::views::flat_repeat_n(value, count);
975  * insert(pos, concatenated.cbegin(), concatenated.cend())
976  */
977 
978  size_type value_len = 0;
979  if constexpr (std::ranges::sized_range<rng_type>)
980  value_len = std::ranges::size(value);
981  else
982  value_len = std::distance(std::ranges::begin(value), std::ranges::end(value));
983 
984  data_values.reserve(data_values.size() + count * value_len);
985  auto placeholder = views::repeat_n(std::ranges::range_value_t<rng_type>{}, count * value_len)
986  | std::views::common;
987  // insert placeholder so the tail is moved once:
988  data_values.insert(data_values.begin() + data_delimiters[pos_as_num],
989  std::ranges::begin(placeholder),
990  std::ranges::end(placeholder));
991 
992  // assign the actual values to the placeholder:
993  size_t i = data_delimiters[pos_as_num];
994  for (size_t j = 0; j < count; ++j)
995  for (auto && v : value)
996  data_values[i++] = v;
997 
998  data_delimiters.reserve(data_values.size() + count);
999  data_delimiters.insert(data_delimiters.begin() + pos_as_num,
1000  count,
1001  *(data_delimiters.begin() + pos_as_num));
1002 
1003  // adapt delimiters of inserted
1004  for (size_type i = 0; i < count; ++i)
1005  data_delimiters[pos_as_num + i + 1] += value_len * (i + 1);
1006 
1007  // adapt delimiters after that
1008  // TODO parallel execution policy or vectorization?
1009  std::for_each(data_delimiters.begin() + pos_as_num + count + 1,
1010  data_delimiters.end(),
1011  [full_len = value_len * count] (auto & d) { d += full_len; });
1012 
1013  return begin() + pos_as_num;
1014  }

1041  iterator insert(const_iterator pos, begin_iterator_type first, end_iterator_type last)
1043  requires iter_value_t_is_compatible_with_value_type<begin_iterator_type> &&
1044  std::sized_sentinel_for<end_iterator_type, begin_iterator_type>
1046  {
1047  auto const pos_as_num = std::distance(cbegin(), pos);
1048  // TODO SEQAN_UNLIKELY
1049  if (last - first == 0)
1050  return begin() + pos_as_num;
1051 
1052  auto const ilist =
1053  std::ranges::subrange<begin_iterator_type, end_iterator_type>(first,
1054  last,
1055  std::ranges::distance(first, last));
1056 
1057  data_delimiters.reserve(data_values.size() + ilist.size());
1058  data_delimiters.insert(data_delimiters.begin() + pos_as_num,
1059  ilist.size(),
1060  *(data_delimiters.begin() + pos_as_num));
1061 
1062 
1063  // adapt delimiters of inserted region
1064  size_type full_len = 0;
1065  for (size_type i = 0; i < ilist.size(); ++i, ++first)

1067  full_len += std::ranges::distance(*first);
1068  data_delimiters[pos_as_num + 1 + i] += full_len;
1069  }
1070 
1071  // adapt values of inserted region
1072  auto placeholder = views::repeat_n(std::ranges::range_value_t<value_type>{}, full_len)
1073  | std::views::common;
1074  // insert placeholder so the tail is moved only once:
1075  data_values.insert(data_values.begin() + data_delimiters[pos_as_num],
1076  std::ranges::begin(placeholder),
1077  std::ranges::end(placeholder));
