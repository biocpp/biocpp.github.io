846 template <std::ranges::forward_range rng_type>
847 iterator insert(const_iterator pos, size_type const count, rng_type && value)
849 requires is_compatible_with_value_type<rng_type>
851 {
852 auto const pos_as_num = std::distance(cbegin(), pos); // we want to insert BEFORE this position
853 // TODO SEQAN_UNLIKELY
854 if (count == 0)
855 return begin() + pos_as_num;
856
857 /* TODO implement views::flat_repeat_n that is like
858 * views::repeat_n(value, count) | views::join | ranges::views::bounded;
859 * but preserves random access and size.
860 *
861 * then do
862 * auto concatenated = ranges::views::flat_repeat_n(value, count);
863 * insert(pos, concatenated.cbegin(), concatenated.cend())
864 */
865
866 size_type value_len = 0;
867 if constexpr (std::ranges::sized_range<rng_type>)
868 value_len = std::ranges::size(value);
869 else
870 value_len = std::distance(std::ranges::begin(value), std::ranges::end(value));
871
872 data_values.reserve(data_values.size() + count * value_len);
873 auto placeholder =
874 views::repeat_n(std::ranges::range_value_t<rng_type>{}, count * value_len) | std::views::common;
875 // insert placeholder so the tail is moved once:
876 data_values.insert(data_values.begin() + data_delimiters[pos_as_num],
877 std::ranges::begin(placeholder),
878 std::ranges::end(placeholder));
879
880 // assign the actual values to the placeholder:
881 size_t i = data_delimiters[pos_as_num];
882 for (size_t j = 0; j < count; ++j)
883 for (auto && v : value)
884 data_values[i++] = v;
885
886 data_delimiters.reserve(data_values.size() + count);
887 data_delimiters.insert(data_delimiters.begin() + pos_as_num, count, *(data_delimiters.begin() + pos_as_num));
888
889 // adapt delimiters of inserted
890 for (size_type i = 0; i < count; ++i)
891 data_delimiters[pos_as_num + i + 1] += value_len * (i + 1);
892
893 // adapt delimiters after that
894 // TODO parallel execution policy or vectorization?
895 std::for_each(data_delimiters.begin() + pos_as_num + count + 1,
896 data_delimiters.end(),
897 [full_len = value_len * count](auto & d) { d += full_len; });
898
899 return begin() + pos_as_num;
900 }

926 template <std::forward_iterator begin_iterator_type, typename end_iterator_type>
927 iterator insert(const_iterator pos, begin_iterator_type first, end_iterator_type last)
929 requires(iter_value_t_is_compatible_with_value_type<begin_iterator_type> &&
930 std::sized_sentinel_for<end_iterator_type, begin_iterator_type>)
932 {
933 auto const pos_as_num = std::distance(cbegin(), pos);
934 // TODO SEQAN_UNLIKELY
935 if (last - first == 0)
936 return begin() + pos_as_num;
937
938 auto const ilist =
939 std::ranges::subrange<begin_iterator_type, end_iterator_type>(first,
940 last,
941 std::ranges::distance(first, last));
942
943 data_delimiters.reserve(data_values.size() + ilist.size());
944 data_delimiters.insert(data_delimiters.begin() + pos_as_num,
945 ilist.size(),
946 *(data_delimiters.begin() + pos_as_num));
947
948 // adapt delimiters of inserted region
949 size_type full_len = 0;
950 for (size_type i = 0; i < ilist.size(); ++i, ++first)
951 {
952 full_len += std::ranges::distance(*first);
953 data_delimiters[pos_as_num + 1 + i] += full_len;
954 }
955
956 // adapt values of inserted region
957 auto placeholder = views::repeat_n(std::ranges::range_value_t<value_type>{}, full_len) | std::views::common;
958 // insert placeholder so the tail is moved only once:
959 data_values.insert(data_values.begin() + data_delimiters[pos_as_num],
960 std::ranges::begin(placeholder),
961 std::ranges::end(placeholder));
962
963 // assign the actual values to the placeholder:
964 size_t i = data_delimiters[pos_as_num];
965 for (auto && v0 : ilist)
966 for (auto && v1 : v0)
967 data_values[i++] = v1;
968
969 // adapt delimiters behind inserted region
970 // TODO parallel execution policy or vectorization?
971 std::for_each(data_delimiters.begin() + pos_as_num + ilist.size() + 1,
972 data_delimiters.end(),
973 [full_len](auto & d) { d += full_len; });
974
975 return begin() + pos_as_num;
976 }

1024 iterator erase(const_iterator first, const_iterator last)
1025 {
1026 auto const dist = std::distance(cbegin(), last);
1027 // TODO SEQAN_UNLIKELY
1028 if (last - first == 0)
1029 return begin() + dist;
1030
1031 auto const distf = std::distance(cbegin(), first);
1032
1033 // we need to scan once over the input
1034 size_type sum_size{0};
1035 for (; first != last; ++first)
1036 sum_size += std::ranges::size(*first);
1037
1038 data_values.erase(data_values.begin() + data_delimiters[distf], data_values.begin() + data_delimiters[dist]);
1039
1040 data_delimiters.erase(data_delimiters.begin() + distf + 1, data_delimiters.begin() + dist + 1);
1041
1042 // adapt delimiters after that
1043 // TODO parallel execution policy or vectorization?
1044 std::for_each(data_delimiters.begin() + distf + 1,
1045 data_delimiters.end(),
1046 [sum_size](auto & d) { d -= sum_size; });
1047 return begin() + dist;
1048 }
