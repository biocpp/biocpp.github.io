845  template <std::ranges::forward_range rng_type>
846  iterator insert(const_iterator pos, size_type const count, rng_type && value)
848  requires is_compatible_with_value_type<rng_type>
850  {
851  auto const pos_as_num = std::distance(cbegin(), pos); // we want to insert BEFORE this position
852  // TODO SEQAN_UNLIKELY
853  if (count == 0)
854  return begin() + pos_as_num;
855 
856  /* TODO implement views::flat_repeat_n that is like
857  * views::repeat_n(value, count) | views::join | ranges::views::bounded;
858  * but preserves random access and size.
859  *
860  * then do
861  * auto concatenated = ranges::views::flat_repeat_n(value, count);
862  * insert(pos, concatenated.cbegin(), concatenated.cend())
863  */
864 
865  size_type value_len = 0;
866  if constexpr (std::ranges::sized_range<rng_type>)
867  value_len = std::ranges::size(value);
868  else
869  value_len = std::distance(std::ranges::begin(value), std::ranges::end(value));
870 
871  data_values.reserve(data_values.size() + count * value_len);
872  auto placeholder =
873  views::repeat_n(std::ranges::range_value_t<rng_type>{}, count * value_len) | std::views::common;
874  // insert placeholder so the tail is moved once:
875  data_values.insert(data_values.begin() + data_delimiters[pos_as_num],
876  std::ranges::begin(placeholder),
877  std::ranges::end(placeholder));
878 
879  // assign the actual values to the placeholder:
880  size_t i = data_delimiters[pos_as_num];
881  for (size_t j = 0; j < count; ++j)
882  for (auto && v : value)
883  data_values[i++] = v;
884 
885  data_delimiters.reserve(data_values.size() + count);
886  data_delimiters.insert(data_delimiters.begin() + pos_as_num, count, *(data_delimiters.begin() + pos_as_num));
887 
888  // adapt delimiters of inserted
889  for (size_type i = 0; i < count; ++i)
890  data_delimiters[pos_as_num + i + 1] += value_len * (i + 1);
891 
892  // adapt delimiters after that
893  // TODO parallel execution policy or vectorization?
894  std::for_each(data_delimiters.begin() + pos_as_num + count + 1,
895  data_delimiters.end(),
896  [full_len = value_len * count](auto & d) { d += full_len; });
897 
898  return begin() + pos_as_num;
899  }

925  template <std::forward_iterator begin_iterator_type, typename end_iterator_type>
926  iterator insert(const_iterator pos, begin_iterator_type first, end_iterator_type last)
928  requires(iter_value_t_is_compatible_with_value_type<begin_iterator_type> &&
929  std::sized_sentinel_for<end_iterator_type, begin_iterator_type>)
931  {
932  auto const pos_as_num = std::distance(cbegin(), pos);
933  // TODO SEQAN_UNLIKELY
934  if (last - first == 0)
935  return begin() + pos_as_num;
936 
937  auto const ilist =
938  std::ranges::subrange<begin_iterator_type, end_iterator_type>(first,
939  last,
940  std::ranges::distance(first, last));
941 
942  data_delimiters.reserve(data_values.size() + ilist.size());
943  data_delimiters.insert(data_delimiters.begin() + pos_as_num,
944  ilist.size(),
945  *(data_delimiters.begin() + pos_as_num));
946 
947  // adapt delimiters of inserted region
948  size_type full_len = 0;
949  for (size_type i = 0; i < ilist.size(); ++i, ++first)
950  {
951  full_len += std::ranges::distance(*first);
952  data_delimiters[pos_as_num + 1 + i] += full_len;
953  }
954 
955  // adapt values of inserted region
956  auto placeholder = views::repeat_n(std::ranges::range_value_t<value_type>{}, full_len) | std::views::common;
957  // insert placeholder so the tail is moved only once:
958  data_values.insert(data_values.begin() + data_delimiters[pos_as_num],
959  std::ranges::begin(placeholder),
960  std::ranges::end(placeholder));
961 
962  // assign the actual values to the placeholder:
963  size_t i = data_delimiters[pos_as_num];
964  for (auto && v0 : ilist)
965  for (auto && v1 : v0)
966  data_values[i++] = v1;
967 
968  // adapt delimiters behind inserted region
969  // TODO parallel execution policy or vectorization?
970  std::for_each(data_delimiters.begin() + pos_as_num + ilist.size() + 1,
971  data_delimiters.end(),
972  [full_len](auto & d) { d += full_len; });
973 
974  return begin() + pos_as_num;
975  }
