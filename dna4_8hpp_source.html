99 // reverse mapping for characters and their lowercase
100 for (size_t rnk = 0u; rnk < alphabet_size; ++rnk)
101 {
102 ret[rank_to_char[rnk]] = rnk;
103 ret[detail::to_lower(rank_to_char[rnk])] = rnk;

108 ret['u'] = ret['t'];
