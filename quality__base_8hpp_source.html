112 static constexpr std::array<rank_type, 256> phred_to_rank = []() constexpr
113 {
114 std::array<rank_type, 256> ret{};
115
116 for (int64_t i = std::numeric_limits<phred_type>::lowest(); i <= std::numeric_limits<phred_type>::max(); ++i)
117 {
118 if (i < derived_type::offset_phred) // map too-small to smallest possible
119 ret[static_cast<rank_type>(i)] = 0;
120 else if (i >= derived_type::offset_phred +
121 static_cast<int64_t>(alphabet_size)) // map too-large to highest possible
122 ret[static_cast<rank_type>(i)] = alphabet_size - 1;
123 else // map valid range to identity
124 ret[static_cast<rank_type>(i)] = i - derived_type::offset_phred;
125 }
126 return ret;
127 }
