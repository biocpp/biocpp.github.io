374 template <size_t N>
375 constexpr void assign(char const (&lit)[N])
376 {
377 static_assert(N <= bit_capacity + 1, "Length of string literal exceeds capacity of dynamic_bitset.");
378 assert(lit[N - 1] == '\0');
379 uint64_t value{};
380
381 for (size_t i = 0; i != N - 1; ++i)
382 {
383 if (lit[i] == '0')
384 {
385 value <<= 1;
386 }
387 else if (lit[i] == '1')
388 {
389 value <<= 1;
390 value |= 1u;
391 }
392 else
393 {
394 throw std::invalid_argument{"The string to construct a dynamic_bitset from may only contain 0 and 1."};
395 }
396 }
397
398 *this = value;
399 resize(N - 1);
400 }

1226 template <std::forward_iterator begin_it_type, typename end_it_type>
1228 requires(std::sentinel_for<end_it_type, begin_it_type> &&
1229 std::constructible_from<value_type, std::iter_reference_t<begin_it_type>>)
1231 constexpr iterator insert(const_iterator pos, begin_it_type begin_it, end_it_type end_it) noexcept
1232 {
1233 auto const pos_as_num = std::ranges::distance(cbegin(), pos);
1234 auto const length = std::ranges::distance(begin_it, end_it);
1235
1236 if (length == 0)
1237 return begin(); // nothing to insert
1238
1239 size_type const tmp_size{size()};
1240 resize(tmp_size + length);
1241
1242 for (size_type i = tmp_size + length - 1; i > pos_as_num + length - 1; --i)
1243 (*this)[i] = (*this)[i - length];
1244
1245 // std::ranges::copy(begin_it, end_it, (*this)[pos_as_num]);
1246 for (auto i = pos_as_num; begin_it != end_it; ++i, ++begin_it)
1247 (*this)[i] = *begin_it;
1248
1249 return begin() + pos_as_num;
1250 }
