69  static_assert(bits_per_letter <= 64, "alphabet must be representable in at most 64bit.");

72  using word_type = uint64_t;
74  static constexpr size_t word_size = sizeof(word_type) * CHAR_BIT;
76  static constexpr size_t letters_per_word = word_size / bits_per_letter;
78  static constexpr uint64_t mask = (1ull << bits_per_letter) - 1ull;

90  static uint64_t get_rank(data_type const & vec, size_t const i) noexcept
91  {
92  assert(i / letters_per_word < vec.size());
93  uint64_t const word = vec[i / letters_per_word];
94  size_t const offset = (i % letters_per_word) * bits_per_letter;
95  return (word >> offset) & mask;
96  }
97 
99  static void set_rank(data_type & vec, size_t const i, uint64_t const rank) noexcept
100  {
101  assert(i / letters_per_word < vec.size());
102  uint64_t & word = vec[i / letters_per_word];
103  size_t const offset = (i % letters_per_word) * bits_per_letter;
104  word &= ~(mask << offset);
105  word |= rank << offset;
106  }
107 
109  void clear_unused_bits_in_last_word()
110  {
111  if (data.empty() || (size() % letters_per_word == 0))
112  return;
113 
114  size_t const offset = (size() % letters_per_word) * bits_per_letter;
115  data.back() &= ((1ull << offset) - 1ull);
116  }
