429template <bool exactly, bool or_throw>
430struct take_fn
431{
433 constexpr auto operator()(size_t const size) const { return adaptor_from_functor{*this, size}; }
434
438 template <std::ranges::range urng_t>
439 constexpr auto operator()(urng_t && urange, size_t target_size) const
440 {
441 static_assert(std::ranges::viewable_range<urng_t>,
442 "The views::take adaptor can only be passed viewable_ranges, i.e. Views or &-to-non-View.");
443
444 // safeguard against wrong size
445 if constexpr (std::ranges::sized_range<urng_t>)
446 {
447 if constexpr (or_throw)
448 {
449 if (target_size > std::ranges::size(urange))
450 {
451 throw std::invalid_argument{
452 "You are trying to construct a views::take_exactly_or_throw from a "
453 "range that is strictly smaller."};
454 }
455 }
456 else
457 {
458 target_size = std::min<size_t>(target_size, std::ranges::size(urange));
459 }
460 }
461
462 // string_view
463 if constexpr (meta::is_type_specialisation_of_v<std::remove_cvref_t<urng_t>, std::basic_string_view>)
464 {
465 return urange.substr(0, target_size);
466 }
467 // string const &
468 else if constexpr (meta::is_type_specialisation_of_v<std::remove_cvref_t<urng_t>, std::basic_string> &&
469 std::is_const_v<std::remove_reference_t<urng_t>>)
470 {
471 return std::basic_string_view{std::ranges::data(urange), target_size};
472 }
473 // contiguous
474 else if constexpr (std::ranges::borrowed_range<urng_t> && std::ranges::contiguous_range<urng_t> &&
475 std::ranges::sized_range<urng_t>)
476 {
477 return std::span{std::ranges::data(urange), target_size};
478 }
479 // random_access
480 else if constexpr (std::ranges::borrowed_range<urng_t> && std::ranges::random_access_range<urng_t> &&
481 std::ranges::sized_range<urng_t>)
482 {
483 return std::ranges::subrange<std::ranges::iterator_t<urng_t>, std::ranges::iterator_t<urng_t>>{
484 std::ranges::begin(urange),
485 std::ranges::begin(urange) + target_size,
486 target_size};
487 }
488 // our type
489 else
490 {
491 return view_take<std::views::all_t<urng_t>, exactly, or_throw>{std::forward<urng_t>(urange), target_size};
492 }
493 }
494};
