572  template <size_t index>
573  constexpr rank_type to_component_rank() const noexcept
574  {
575  if constexpr (alphabet_size < 1024) // computation is cached for small alphabets
576  {
577  return rank_to_component_rank[index][to_rank()];
578  }
579  else
580  {
581  return (to_rank() / cummulative_alph_sizes[index]) %
582  bio::alphabet_size<pack_traits::at<index, component_types...>>;
583  }
584  }
585 
587  template <size_t index>
588  constexpr void assign_component_rank(ptrdiff_t const r) noexcept
589  {
590  assign_rank(static_cast<ptrdiff_t>(to_rank()) +
591  ((r - static_cast<ptrdiff_t>(to_component_rank<index>())) *
592  static_cast<ptrdiff_t>(cummulative_alph_sizes[index])));
593  }
594 
596  static constexpr std::array<rank_type, component_list::size()> cummulative_alph_sizes
597  {
598  [] () constexpr
599  {
600  // create array (1, |sigma1|, |sigma1|*|sigma2|, ... , |sigma1|*...*|sigmaN|)
601  std::array<rank_type, component_list::size() + 1> ret{};
602  ret[0] = 1;
603  size_t count = 1;
604  using reverse_list_t = decltype(bio::list_traits::detail::reverse(component_list{}));
605  bio::detail::for_each<reverse_list_t>([&] (auto alphabet_type_identity) constexpr
606  {
607  using alphabet_t = typename decltype(alphabet_type_identity)::type;
608  ret[count] = static_cast<rank_type>(bio::alphabet_size<alphabet_t> * ret[count - 1]);
609  ++count;
610  });
611 
612  // reverse and strip one: (|sigma1|*...*|sigmaN-1|, ... |sigma1|*|sigma2|, |sigma1|, 1)
613  // reverse order guarantees that the first alphabet is the most significant rank contributer
614  // resulting in element-wise alphabetical ordering on comparison
615  std::array<rank_type, component_list::size()> ret2{};
616  for (size_t i = 0; i < component_list::size(); ++i)
617  ret2[i] = ret[component_list::size() - i - 1];
618 
619  return ret2;
620  }()
621  };
622 
624  template <std::size_t ...idx>
625  static constexpr rank_type rank_sum_helper(component_types ... components, std::index_sequence<idx...> const &) noexcept
626  {
627  return ((bio::to_rank(components) * cummulative_alph_sizes[idx]) + ...);
628  }
629 
631  static constexpr std::array<std::array<rank_type, alphabet_size < 1024 ? alphabet_size : 0>, // not for big alphs
632  list_traits::size<component_list>> rank_to_component_rank
633  {
634  [] () constexpr
635  {
636  std::array<std::array<rank_type, alphabet_size < 1024 ? alphabet_size : 0>, // not for big alphs
637  list_traits::size<component_list>> ret{};
638 
639  if constexpr (alphabet_size < 1024)
640  {
641  std::array<size_t, alphabet_size> alph_sizes{ bio::alphabet_size<component_types>... };
642 
643  for (size_t i = 0; i < list_traits::size<component_list>; ++i)
644  for (size_t j = 0; j < static_cast<size_t>(alphabet_size); ++j)
645  ret[i][j] = (j / cummulative_alph_sizes[i]) % alph_sizes[i];
646  }
647 
648  return ret;
649  }()
650  };
