491  template <size_t index>
492  constexpr rank_type to_component_rank() const noexcept
493  {
494  if constexpr (alphabet_size < 1024) // computation is cached for small alphabets
495  {
496  return rank_to_component_rank[index][to_rank()];
497  }
498  else
499  {
500  return (to_rank() / cummulative_alph_sizes[index]) %
501  bio::alphabet::size<meta::detail::pack_traits::at<index, component_types...>>;
502  }
503  }
504 
506  template <size_t index>
507  constexpr void assign_component_rank(ptrdiff_t const r) noexcept
508  {
509  assign_rank(static_cast<ptrdiff_t>(to_rank()) + ((r - static_cast<ptrdiff_t>(to_component_rank<index>())) *
510  static_cast<ptrdiff_t>(cummulative_alph_sizes[index])));
511  }
512 
514  static constexpr std::array<rank_type, component_list::size()> cummulative_alph_sizes = []() constexpr
515  {
516  // create array (1, |sigma1|, |sigma1|*|sigma2|, ... , |sigma1|*...*|sigmaN|)
517  std::array<rank_type, component_list::size() + 1> ret{};
518  ret[0] = 1;
519  size_t count = 1;
520  using reverse_list_t = decltype(meta::list_traits::detail::reverse(component_list{}));
521  bio::meta::detail::for_each<reverse_list_t>([&](auto alphabet_type_identity) constexpr {
522  using alphabet_t = typename decltype(alphabet_type_identity)::type;
523  ret[count] = static_cast<rank_type>(bio::alphabet::size<alphabet_t> * ret[count - 1]);
524  ++count;
525  });
526 
527  // reverse and strip one: (|sigma1|*...*|sigmaN-1|, ... |sigma1|*|sigma2|, |sigma1|, 1)
528  // reverse order guarantees that the first alphabet is the most significant rank contributer
529  // resulting in element-wise alphabetical ordering on comparison
530  std::array<rank_type, component_list::size()> ret2{};
531  for (size_t i = 0; i < component_list::size(); ++i)
532  ret2[i] = ret[component_list::size() - i - 1];
533 
534  return ret2;
535  }
536  ();

539  template <std::size_t... idx>
540  static constexpr rank_type rank_sum_helper(component_types... components,
541  std::index_sequence<idx...> const &) noexcept
542  {
543  return ((bio::alphabet::to_rank(components) * cummulative_alph_sizes[idx]) + ...);
544  }
545 
547  static constexpr std::array < std::array<rank_type,
548  alphabet_size<1024 ? alphabet_size : 0>, // not for big alphs
549  meta::list_traits::size<component_list>>
550  rank_to_component_rank = []() constexpr
551  {
552  std::array < std::array<rank_type,
553  alphabet_size<1024 ? alphabet_size : 0>, // not for big alphs
554  meta::list_traits::size<component_list>>
555  ret{};
556 
557  if constexpr (alphabet_size < 1024)
558  {
559  std::array<size_t, alphabet_size> alph_sizes{bio::alphabet::size<component_types>...};
560 
561  for (size_t i = 0; i < meta::list_traits::size<component_list>; ++i)
562  for (size_t j = 0; j < static_cast<size_t>(alphabet_size); ++j)
563  ret[i][j] = (j / cummulative_alph_sizes[i]) % alph_sizes[i];
564  }
565 
566  return ret;
567  }
568  ();
