602  std::constructible_from<value_type, std::iter_reference_t<begin_it_type>>)
604  constexpr iterator insert(const_iterator pos, begin_it_type begin_it, end_it_type end_it) noexcept(is_noexcept)
605  {
606  auto const pos_as_num = std::ranges::distance(cbegin(), pos);
607  auto const length = std::ranges::distance(begin_it, end_it);
608 
609  assert(pos_as_num + length <= capacity());
610 
611  if (length == 0)
612  return begin(); // nothing to insert
613 
614  for (size_type i = sz + length - 1; i > pos_as_num + length - 1; --i)
615  data_[i] = data_[i - length];
616 
617  std::ranges::copy(begin_it, end_it, &data_[pos_as_num]);
618  sz += length;
619  return begin() + pos_as_num;
620  }
