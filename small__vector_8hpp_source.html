598 template <std::forward_iterator begin_it_type, typename end_it_type>
600 requires(std::sentinel_for<end_it_type, begin_it_type> &&
601 std::constructible_from<value_type, std::iter_reference_t<begin_it_type>>)
603 constexpr iterator insert(const_iterator pos, begin_it_type begin_it, end_it_type end_it) noexcept(is_noexcept)
604 {
605 auto const pos_as_num = std::ranges::distance(cbegin(), pos);
606 auto const length = std::ranges::distance(begin_it, end_it);
607
608 assert(pos_as_num + length <= capacity());
609
610 if (length == 0)
611 return begin(); // nothing to insert
612
613 for (size_type i = sz + length - 1; i > pos_as_num + length - 1; --i)
614 data_[i] = data_[i - length];
615
616 std::ranges::copy(begin_it, end_it, &data_[pos_as_num]);
617 sz += length;
618 return begin() + pos_as_num;
619 }

659 constexpr iterator erase(const_iterator begin_it, const_iterator end_it) noexcept
660 {
661 if (begin_it >= end_it) // [[unlikely]]
662 return begin() + std::ranges::distance(cbegin(), end_it);
663
664 size_type const length = std::ranges::distance(begin_it, end_it);
665 auto out_it = begin() + std::ranges::distance(cbegin(), begin_it);
666
667 while (end_it != cend())
668 *(out_it++) = *(end_it++);
669
670 sz -= length;
671 return begin() + std::ranges::distance(cbegin(), begin_it);
672 }
