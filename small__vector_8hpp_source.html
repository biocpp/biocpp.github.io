645  template <std::forward_iterator begin_it_type, typename end_it_type>
647  requires std::sentinel_for<end_it_type, begin_it_type> &&
648  std::constructible_from<value_type, std::iter_reference_t<begin_it_type>>
650  constexpr iterator insert(const_iterator pos, begin_it_type begin_it, end_it_type end_it) noexcept(is_noexcept)
651  {
652  auto const pos_as_num = std::ranges::distance(cbegin(), pos);
653  auto const length = std::ranges::distance(begin_it, end_it);
654 
655  assert(pos_as_num + length <= capacity());
656 
657  if (length == 0)
658  return begin(); // nothing to insert
659 
660  for (size_type i = sz + length - 1; i > pos_as_num + length - 1; --i)
661  data_[i] = data_[i - length];
662 
663  std::ranges::copy(begin_it, end_it, &data_[pos_as_num]);
664  sz += length;
665  return begin() + pos_as_num;
666  }
