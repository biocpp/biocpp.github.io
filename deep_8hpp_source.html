183 template <typename first_arg_t, typename... stored_arg_types>
185 requires(!std::ranges::input_range<first_arg_t>)
187 constexpr auto operator()(first_arg_t && first, stored_arg_types &&... args) const
188 {
189 // The adaptor currently wrapped is a proto-adaptor and this function has the arguments to "complete" it.
190 // We extract the adaptor that is stored and invoke it with the given arguments.
191 // This returns an adaptor closure object.
192 auto adaptor_closure =
193 std::get<0>(this->arguments)(std::forward<first_arg_t>(first), std::forward<stored_arg_types>(args)...);
194 // Now we wrap this closure object back into a views::deep to get the deep behaviour.
195 return deep<decltype(adaptor_closure)>{std::move(adaptor_closure)};
196 }
197
199 constexpr auto operator()() const
200 {
201 // Proto-adaptors require arguments by definition, but some support defaulting those (e.g. views::translate).
202 // This extracts the proto adaptor and invokes it without args which yields a different object, the closure
203 // with defaulted arguments.
204 auto adaptor_closure = std::get<0>(this->arguments)();
205 // Now we wrap this closure object back into a views::deep to get the deep behaviour.
206 return deep<decltype(adaptor_closure)>{std::move(adaptor_closure)};
207 }
