186  template <typename first_arg_t, typename ...stored_arg_types>
188  requires (!std::ranges::input_range<first_arg_t>)
190  constexpr auto operator()(first_arg_t && first, stored_arg_types && ...args) const
191  {
192  // The adaptor currently wrapped is a proto-adaptor and this function has the arguments to "complete" it.
193  // We extract the adaptor that is stored and invoke it with the given arguments.
194  // This returns an adaptor closure object.
195  auto adaptor_closure = std::get<0>(this->arguments)(std::forward<first_arg_t>(first),
196  std::forward<stored_arg_types>(args)...);
197  // Now we wrap this closure object back into a views::deep to get the deep behaviour.
198  return deep<decltype(adaptor_closure)>{std::move(adaptor_closure)};
199  }
