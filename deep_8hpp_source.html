182 template <typename first_arg_t, typename... stored_arg_types>
184 requires(!std::ranges::input_range<first_arg_t>)
186 constexpr auto operator()(first_arg_t && first, stored_arg_types &&... args) const
187 {
188 // The adaptor currently wrapped is a proto-adaptor and this function has the arguments to "complete" it.
189 // We extract the adaptor that is stored and invoke it with the given arguments.
190 // This returns an adaptor closure object.
191 auto adaptor_closure =
192 std::get<0>(this->arguments)(std::forward<first_arg_t>(first), std::forward<stored_arg_types>(args)...);
193 // Now we wrap this closure object back into a views::deep to get the deep behaviour.
194 return deep<decltype(adaptor_closure)>{std::move(adaptor_closure)};
195 }
196
198 constexpr auto operator()() const
199 {
200 // Proto-adaptors require arguments by definition, but some support defaulting those (e.g. views::translate).
201 // This extracts the proto adaptor and invokes it without args which yields a different object, the closure
202 // with defaulted arguments.
203 auto adaptor_closure = std::get<0>(this->arguments)();
204 // Now we wrap this closure object back into a views::deep to get the deep behaviour.
205 return deep<decltype(adaptor_closure)>{std::move(adaptor_closure)};
206 }
