33template <typename other_t, typename... alternative_types>
34inline constexpr bool variant_general_guard =
35 ((!std::same_as<other_t, alphabet_variant<alternative_types...>>)&&(
36 !std::is_base_of_v<alphabet_variant<alternative_types...>,
37 other_t>)&&(!(std::same_as<other_t, alternative_types> || ...)) &&
38 (!meta::list_traits::contains<alphabet_variant<alternative_types...>, recursive_required_types_t<other_t>>));
39
41template <typename lhs_t, typename rhs_t, bool lhs_rhs_switched, typename... alternative_types>
42inline constexpr bool variant_comparison_guard =
43 (meta::detail::instantiate_if_v<
44 meta::detail::lazy<weakly_equality_comparable_with_trait, rhs_t, alternative_types>,
45 (std::same_as<lhs_t, alphabet_variant<alternative_types...>>)&&(
46 variant_general_guard<rhs_t, alternative_types...>)&&!(lhs_rhs_switched &&
47 meta::is_type_specialisation_of_v<rhs_t,
48 alphabet_variant>)> ||
49 ...);

404 template <typename alphabet_variant_t, typename indirect_alternative_type>
405 friend constexpr auto operator==(alphabet_variant_t const lhs, indirect_alternative_type const rhs) noexcept
406 -> std::enable_if_t<
407 detail::variant_comparison_guard<alphabet_variant_t, indirect_alternative_type, false, alternative_types...>,
408 bool>
409 {
410 using alternative_predicate = detail::weakly_equality_comparable_with_<indirect_alternative_type>;

413 using alternative_t = meta::list_traits::at<alternative_position, alternatives>;
414 return lhs.template holds_alternative<alternative_t>() &&

465 template <size_t index, bool throws>
466 constexpr auto convert_impl() const noexcept(!throws) -> meta::list_traits::at<index, alternatives>
467 {
468 static_assert(index < alphabet_size, "The alphabet_variant contains less alternatives than you are checking.");
469 using alternative_t = meta::list_traits::at<index, alternatives>;
470
471 if constexpr (throws)
472 {
473 if (!holds_alternative<index>()) // [[unlikely]]
474 {
475 throw std::bad_variant_access{};
476 }
477 }
478
479 return bio::alphabet::assign_rank_to(to_rank() - partial_sum_sizes[index], alternative_t{});
480 }

489 static constexpr std::array<rank_type, sizeof...(alternative_types) + 1> partial_sum_sizes = []() constexpr
490 {
491 constexpr size_t N = sizeof...(alternative_types) + 1;
492
493 std::array<rank_type, N> partial_sum{0, bio::alphabet::size<alternative_types>...};
494 for (size_t i = 1u; i < N; ++i)
495 partial_sum[i] += partial_sum[i - 1];
496
497 return partial_sum;
498 }

507 static constexpr std::array<char_type, alphabet_size> rank_to_char = []() constexpr
508 {
509 // Explicitly writing assign_rank_to_char within assign_rank_to_char
510 // causes this bug (g++-7 and g++-8):
511 // https://gcc.gnu.org/bugzilla/show_bug.cgi?id=84684
512 auto assign_rank_to_char = [](auto alternative, size_t rank) constexpr
513 {
514 return bio::alphabet::to_char(bio::alphabet::assign_rank_to(rank, alternative));
515 };
516
517 auto assign_value_to_char = [assign_rank_to_char](auto alternative,
518 auto & value_to_char,
519 auto & value) constexpr
520 {
521 using alternative_t = std::decay_t<decltype(alternative)>;
522 for (size_t i = 0u; i < bio::alphabet::size<alternative_t>; ++i, ++value)
523 value_to_char[value] = assign_rank_to_char(alternative, i);
524 };
525
526 unsigned value = 0u;
527 std::array<char_type, alphabet_size> value_to_char{};
528
529 // initializer lists guarantee sequencing;
530 // the following expression behaves as:
531 // for(auto alternative: alternative_types)
532 // assign_rank_to_char(alternative, rank_to_char, value);
533 ((assign_value_to_char(alternative_types{}, value_to_char, value)), ...);
534
535 return value_to_char;
536 }

569 static constexpr auto first_valid_char_table = []() constexpr
570 {
571 constexpr size_t alternative_size = sizeof...(alternative_types);
572 constexpr size_t table_size = meta::detail::size_in_values_v<char_type>;
573 using first_alphabet_t = meta::detail::min_viable_uint_t<alternative_size>;
574
575 std::array<first_alphabet_t, table_size> lookup_table{};
576
577 for (size_t i = 0u; i < table_size; ++i)
578 {
579 char_type chr = static_cast<char_type>(i);
580
581 std::array<bool, alternative_size> valid_chars{char_is_valid_for<alternative_types>(chr)...};
582
583#if defined(__cpp_lib_constexpr_algorithms) && __cpp_lib_constexpr_algorithms >= 201806L
584 // the following lines only works beginning from c++20
585 auto found_it = std::find(valid_chars.begin(), valid_chars.end(), true);
586 lookup_table[i] = found_it - valid_chars.begin();
587#else
588 size_t found_index = 0u;
589 for (; found_index < valid_chars.size() && !valid_chars[found_index]; ++found_index)
590 ;
591 lookup_table[i] = found_index;
592#endif // defined(__cpp_lib_constexpr_algorithms) && __cpp_lib_constexpr_algorithms >= 201806L
593 }
594
595 return lookup_table;
596 }
597 ();

605 static constexpr std::array<rank_type, meta::detail::size_in_values_v<char_type>> char_to_rank = []() constexpr
606 {
607 constexpr size_t alternative_size = sizeof...(alternative_types);
608 constexpr size_t table_size = meta::detail::size_in_values_v<char_type>;
609
610 std::array<rank_type, table_size> char_to_rank{};
611
612 for (size_t i = 0u; i < table_size; ++i)
613 {
614 char_type chr = static_cast<char_type>(i);
615
616 std::array<rank_type, alternative_size> ranks{rank_by_type_(assign_char_to(chr, alternative_types{}))...};
617
618 // if no char_is_valid_for any alternative use the rank of the first alternative
619 char_to_rank[i] = first_valid_char_table[i] < alternative_size ? ranks[first_valid_char_table[i]] : 0;
620 }
621
622 return char_to_rank;
623 }
