570  static constexpr auto first_valid_char_table
571  {
572  [] () constexpr
573  {
574  constexpr size_t alternative_size = sizeof...(alternative_types);
575  constexpr size_t table_size = detail::size_in_values_v<char_type>;
576  using first_alphabet_t = detail::min_viable_uint_t<alternative_size>;
577 
578  std::array<first_alphabet_t, table_size> lookup_table{};
579 
580  for (size_t i = 0u; i < table_size; ++i)
581  {
582  char_type chr = static_cast<char_type>(i);
583 
584  std::array<bool, alternative_size> valid_chars{char_is_valid_for<alternative_types>(chr)...};
585 
586 #if defined(__cpp_lib_constexpr_algorithms) && __cpp_lib_constexpr_algorithms >= 201806L
587  // the following lines only works beginning from c++20
588  auto found_it = std::find(valid_chars.begin(), valid_chars.end(), true);
589  lookup_table[i] = found_it - valid_chars.begin();
590 #else
591  size_t found_index = 0u;
592  for (; found_index < valid_chars.size() && !valid_chars[found_index]; ++found_index);
593  lookup_table[i] = found_index;
594 #endif // defined(__cpp_lib_constexpr_algorithms) && __cpp_lib_constexpr_algorithms >= 201806L
595  }
596 
597  return lookup_table;
598  }()
