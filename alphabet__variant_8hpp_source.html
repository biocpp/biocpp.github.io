569  static constexpr auto first_valid_char_table = []() constexpr
570  {
571  constexpr size_t alternative_size = sizeof...(alternative_types);
572  constexpr size_t table_size = meta::detail::size_in_values_v<char_type>;
573  using first_alphabet_t = meta::detail::min_viable_uint_t<alternative_size>;
574 
575  std::array<first_alphabet_t, table_size> lookup_table{};
576 
577  for (size_t i = 0u; i < table_size; ++i)
578  {
579  char_type chr = static_cast<char_type>(i);
580 
581  std::array<bool, alternative_size> valid_chars{char_is_valid_for<alternative_types>(chr)...};
582 
583 #if defined(__cpp_lib_constexpr_algorithms) && __cpp_lib_constexpr_algorithms >= 201806L
584  // the following lines only works beginning from c++20
585  auto found_it = std::find(valid_chars.begin(), valid_chars.end(), true);
586  lookup_table[i] = found_it - valid_chars.begin();
587 #else
588  size_t found_index = 0u;
589  for (; found_index < valid_chars.size() && !valid_chars[found_index]; ++found_index)
590  ;
591  lookup_table[i] = found_index;
592 #endif // defined(__cpp_lib_constexpr_algorithms) && __cpp_lib_constexpr_algorithms >= 201806L
593  }
594 
595  return lookup_table;
596  }
