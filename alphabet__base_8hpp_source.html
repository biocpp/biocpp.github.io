53template <typename derived_type, size_t size, typename char_t = char>
54class alphabet_base
55{
56protected:
57 static_assert(size != 0, "alphabet size must be >= 1"); // == 1 is handled below in separate specialisation
58
63 using char_type = std::conditional_t<std::same_as<char_t, void>, char, char_t>;
65 using rank_type = meta::detail::min_viable_uint_t<size - 1>;
67
68public:
72 constexpr alphabet_base() noexcept = default;
73 constexpr alphabet_base(alphabet_base const &) noexcept = default;
74 constexpr alphabet_base(alphabet_base &&) noexcept = default;

97 constexpr char_type to_char() const noexcept
99 requires(!std::same_as<char_t, void>)
101 {
102 return derived_type::rank_to_char[rank];
103 }
104
119 constexpr rank_type to_rank() const noexcept { return rank; }
121
140 constexpr derived_type & assign_char(char_type const c) noexcept
142 requires(!std::same_as<char_t, void>)
144 {
145 using index_t = std::make_unsigned_t<char_type>;
146 rank = derived_type::char_to_rank[static_cast<index_t>(c)];
147 return static_cast<derived_type &>(*this);
148 }
149
165 constexpr derived_type & assign_rank(rank_type const c) noexcept
166 {
167 assert(static_cast<size_t>(c) < static_cast<size_t>(alphabet_size));
168 rank = c;
169 return static_cast<derived_type &>(*this);
170 }
172
174 static constexpr size_t alphabet_size = size;
175
178
180 friend constexpr bool operator==(derived_type const lhs, derived_type const rhs)
181 {
182 return lhs.to_rank() == rhs.to_rank();
183 }
184
186 friend constexpr auto operator<=>(derived_type const lhs, derived_type const rhs)
187 {
188 return lhs.to_rank() <=> rhs.to_rank();
189 }

192private:
194 rank_type rank;
