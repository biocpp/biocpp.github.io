116 {
117 char_type c = static_cast<char_type>(i);
118
119 ret[i] =
120 detail::is_lower(c)
121 ? alphabet::to_rank(assign_char_to(detail::to_upper(c), sequence_alphabet_type{})) + alphabet_size / 2
122 : alphabet::to_rank(assign_char_to(c, sequence_alphabet_type{}));
123 }
