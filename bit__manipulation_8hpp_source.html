69 constexpr size_t next_power_of_two(size_t n)
70 {
71  if (n == 0)

74  --n;

249 template <std::integral type>
250 constexpr type to_little_endian(type const in) noexcept
251 {
252  if constexpr (endian::native == endian::little)
253  {
254  return in;
255  }
256  else if constexpr (endian::native == endian::big)
257  {
258  static_assert(sizeof(type) <= 8,
259  "Can only convert the byte encoding for integral numbers with a size of up to 8 bytes.");
260  static_assert(is_power_of_two(sizeof(type)),
261  "Can only convert the byte encoding for integral numbers whose byte size is a power of two.");
262 
263  if constexpr (sizeof(type) == 2)
264  return htole16(in);
265  else if constexpr (sizeof(type) == 4)
266  return htole32(in);
267  else if constexpr (sizeof(type) == 8)
268  return htole64(in);
269  else

271  }
272  else
273  {
274  static_assert(endian::native == endian::little || endian::native == endian::big,
